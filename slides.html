HTML5 即時簡報聽眾系統

開發團隊: 1KM

指導老師: 黃宇新

演講備忘錄測試

解放你的投影片

透過 HTML5 技術，建造一個演講時給予聽眾和講者互動的即時系統，
使聽眾可以透過簡報和會眾進行交流。
此外，聽眾亦可使用系統取得講者簡報、做筆記或者針對簡報提出問題。

var context;
try {
	window.AudioContext = window.AudioContext || window.webkitAudioContext;	// Fix up for prefixing
	context = new AudioContext();	// Create audio container
	
	$('#keyboard .key').click(function() {
		
		var oscillator = context.createOscillator(); // Create sound source
			oscillator.type = 1; // Square wave
			oscillator.frequency.value = parseInt($(this).attr("f")); // Frequency in hertz
			oscillator.connect(context.destination); // Connect sound to output
		
		// Play instantly
		if (oscillator.start) {
			oscillator.start(0);
		} else if (oscillator.noteOn) {
			oscillator.noteOn(0);
		}
		
		setTimeout(function() {
			oscillator.disconnect();
			oscillator.noteOff(0);
			// Stop instantly
			if (oscillator.stop) {
				oscillator.stop(0);
			} else if (oscillator.noteOff) {
				oscillator.noteOff(0);
			}	
		}, 500);
		
	});
	
} catch(e) {
	$("#keyboard").html("Sorry!
Web Audio API is not supported in this browser!");
}
